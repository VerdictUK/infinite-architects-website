  function generateSafetyHash(input) {
    // Simple hash for audit trail (in production, use crypto)
    let hash = 0;
    const str = input + Date.now().toString();
    for (let i = 0; i < str.length; i++) {
      const char = str.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    return 'HASH_' + Math.abs(hash).toString(16).toUpperCase().padStart(8, '0');
  }
